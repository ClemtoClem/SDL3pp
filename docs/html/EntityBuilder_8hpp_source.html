14// ─────────────────────────────────────────────────────────────────────────────
15// EntityBuilder
16//
17// Dual-component strategy: each entity gets BOTH game-logic components
18// (for our Systems) AND scene-graph components (for SDL3pp rendering).
19//
20// Game components : Transform, Velocity, DirectionComp, SpeedComp,
21// SpriteAnim, CollisionBoxes, HealthComp, EntityTag,
22// + PlayerTag | MobAI | AutoWalk+Dialogue
23//
24// Scene components : Transform2D, GlobalTransform2D, Visible, Tag,
25// AnimatedSprite (frames = 4 dirs × nbCols),
26// SceneGroup, SceneParent, SceneScript (sync script)
27//
28// The SceneScript sync (runs every frame before PropagateTransforms2D):
29// Transform{x,y} (tile-units) → Transform2D.position (pixels)
30// SpriteAnim.scale * dts/ts → Transform2D.scale
31// dir.dir + sa.currentCol → AnimatedSprite.frame
32// Transform.y * 100 → AnimatedSprite.zOrder (Y-depth sort)
33// ─────────────────────────────────────────────────────────────────────────────
34

64 SDL::ECS::EntityId Build(const std::string& typeName,
65 float x, float y, Direction dir,
66 EntityType etype,
67 SDL::ECS::EntityId parentGroup = SDL::ECS::NullEntity) {
68 if (!m_info) return SDL::ECS::NullEntity;
69 auto it = m_info->find(typeName);
70 if (it == m_info->end()) {
71 LOG_GAME(core::LogLevel::Warning) << "EntityBuilder: no config for '" << typeName << "'";
72 return SDL::ECS::NullEntity;
73 }
74 const core::ScriptValue& cfg = it->second;
75
76 // ── Config values ─────────────────────────────────────────────────────
77 std::string img = cfg.Get("texture").AsString("");
78 int tileSize = cfg.Get("tileSize").AsInt(64);
79 int frameSteps = cfg.Get("frameBetweenEachTile").AsInt(5);
80 bool firstIsStand = cfg.Get("firstTileIsStanding").AsBool(true);
81 float scale = cfg.Get("scale").AsFloat(1.5f);
82 float speed = cfg.Get("speed").AsFloat(0.05f);
83 float pushForce = cfg.Get("pushForce").AsFloat(8.f);
84 constexpr int kNbCols = 3; // stand + 2 walk frames
85
86 SDL::TextureRef tex = _LoadTex(img);
87
88 // ── 1. Spawn via SceneBuilder ─────────────────────────────────────────
89 // Creates entity with Transform2D, GlobalTransform2D, Tag, Visible
90 // + AnimatedSprite component
91 float initPx = x * (float)*m_dts;
92 float initPy = y * (float)*m_dts;
93 float initSc = scale * (float)*m_dts / (float)tileSize;
94
95 auto nb = m_scene.AnimSprite2D(typeName, tex)
96 .Position({initPx, initPy})
97 .Scale(initSc)
98 .Play(false) // manual frame selection
99 .ZOrder((int)(y * 100))
100 .Group(etype == EntityType::Player ? "player" :
101 etype == EntityType::Pnj ? "npc" : "mob");
102
103 SDL::ECS::EntityId eid = nb.id;
104
105 // Populate all frames: 4 rows (N,S,W,E) × kNbCols columns
106 if (auto* as = m_world.Get<SDL::ECS::AnimatedSprite>(eid)) {
107 as->frames.clear();
108 as->frames.reserve(4 * kNbCols);
109 for (int row = 0; row < 4; ++row)
110 for (int col = 0; col < kNbCols; ++col)
111 as->frames.push_back({(float)(col * tileSize),
112 (float)(row * tileSize),
113 (float)tileSize, (float)tileSize});
114 as->fps = 0.f; as->playing = false;
115 }
116
117 // Attach to scene hierarchy
118 if (parentGroup != SDL::ECS::NullEntity)
119 nb.AttachTo(parentGroup);
120
121 // ── 2. Game-logic components ──────────────────────────────────────────
122 m_world.Add<Transform>(eid, {x, y});
123 m_world.Add<Velocity>(eid);
124 m_world.Add<DirectionComp>(eid, {dir});
125 m_world.Add<EntityTag>(eid, {etype, typeName});
126 m_world.Add<SpeedComp>(eid, {speed, speed * 2.f, false});
127 m_world.Add<CollisionResult>(eid);
128
129 CollisionBoxes boxes{}; boxes.tileSize = tileSize;
130 _ParseBoxes(cfg, "bodyBox", boxes.body);
131 _ParseBoxes(cfg, "stepBox", boxes.step);
132 m_world.Add<CollisionBoxes>(eid, boxes);
133
134 SpriteAnim sa{};
135 sa.tileSize = tileSize;
136 sa.nbCols = kNbCols;
137 sa.scale = scale;
138 sa.framesPerCol = frameSteps;
139 sa.firstIsStand = firstIsStand;
140 sa.currentCol = firstIsStand ? 0 : 1;
141 m_world.Add<SpriteAnim>(eid, sa);
142
143 HealthComp hc{}; hc.hp = hc.maxHp = 100.f; hc.regenPerSec = 0.5f;
144 m_world.Add<HealthComp>(eid, hc);
145
146 // Type-specific
147 if (etype == EntityType::Player) {
148 m_world.Add<PlayerTag>(eid, {0});
149 } else if (etype == EntityType::Pnj) {
150 m_world.Add<AutoWalk>(eid);
151 m_world.Add<Dialogue>(eid);
152 } else {
153 MobAI ai{};
154 ai.spawnX = x; ai.spawnY = y; ai.pushForce = pushForce;
155 ai.chaseRadius = (etype == EntityType::Dragon) ? 12.f : 8.f;
156 ai.attackRange = (etype == EntityType::Dragon) ? 1.4f : 0.9f;
157 m_world.Add<MobAI>(eid, ai);
158 }

160 // ── 3. Sync script ────────────────────────────────────────────────────
161 // Runs inside scene.Update(dt), before PropagateTransforms2D.
162 // Bridges the gap between game-logic state and scene-graph state.
163 int* dtsPtr = m_dts;
164 nb.OnUpdate([dtsPtr, tileSize](SDL::ECS::EntityId id, SDL::ECS::Context& ctx, float) {
165 auto* tr = ctx.Get<Transform>(id);
166 auto* dir = ctx.Get<DirectionComp>(id);
167 auto* sa = ctx.Get<SpriteAnim>(id);
168 auto* t2d = ctx.Get<SDL::ECS::Transform2D>(id);
169 auto* as = ctx.Get<SDL::ECS::AnimatedSprite>(id);
170 if (!tr || !t2d) return;
171
172 int dts = *dtsPtr;
173 t2d->position = { tr->x * (float)dts, tr->y * (float)dts };
174
175 if (sa) {
176 float sc = sa->scale * (float)dts / (float)tileSize;
177 t2d->scale = {sc, sc};
178 }
179
180 if (dir && sa && as && !as->frames.empty()) {
181 int frame = (int)dir->dir * sa->nbCols + sa->currentCol;
182 as->frame = std::clamp(frame, 0, (int)as->frames.size()-1);
183 as->playing = false;
184 if (tr) as->zOrder = (int)(tr->y * 100);
185 }
186 });
187
188 LOG_GAME(core::LogLevel::Debug)
189 << std::format("Built '{}' eid={} ({:.1f},{:.1f})", typeName, eid, x, y);
190 return eid;
191 }

216 void _ParseBoxes(const core::ScriptValue& cfg, const std::string& key, SDL::FRect out[4]) {
217 SDL::FRect fallback{8,8,48,48};
218 auto v = cfg.Get(key);
219 if (v.IsNull()) { for (int i=0;i<4;++i) out[i]=fallback; return; }
220 if (v.IsRect()) { auto r=v.AsRect(); SDL::FRect b{r.x,r.y,r.w,r.h};
221 for (int i=0;i<4;++i) { out[i]=b; } return; }
222 if (!v.IsSection()) { for (int i=0;i<4;++i) out[i]=fallback; return; }
223 auto d = v.Get("default");
224 SDL::FRect def = fallback;
225 if (d.IsRect()) { auto r=d.AsRect(); def={r.x,r.y,r.w,r.h}; }
226 for (int i=0;i<4;++i) out[i]=def;
227 auto setDir = [&](const char* n, Direction dir) {
228 auto dv = v.Get(n);
229 if (dv.IsRect()) { auto r=dv.AsRect(); out[(int)dir]={r.x,r.y,r.w,r.h}; }
230 };
231 setDir("north",Direction::North); setDir("south",Direction::South);
232 setDir("west",Direction::West); setDir("east",Direction::East);
233 }
