19// ─────────────────────────────────────────────────────────────────────────────
20// GameState — main gameplay
21//
22// Dual-world architecture:
23//
24// m_gameWorld ← ECS::Context shared by game systems AND scene graph
25// Every entity holds:
26// • Game components : Transform, Velocity, SpriteAnim, CollisionBoxes…
27// • Scene components : Transform2D (pixels), AnimatedSprite, SceneScript
28// The SceneScript sync callback bridges game→scene every frame.
29//
30// m_scene ← SceneBuilder (wraps m_gameWorld)
31// Root → entitiesGroup → entity nodes
32// SignalBus for mob-died / teleport / player-spawned
33//
34// m_uiWorld ← separate ECS::Context (UI only)
35// m_ui ← SDL::UI::System
36// HUD row (health bar, polypoints, map name)
37// Canvas widget ← game world rendered here

130 bool HandleEvent(const SDL::Event& ev) override {
131 if (ev.type == SDL::EVENT_QUIT) { m_ctx->quit(); return false; }
132
133 // Forward to UI and scene input handlers
134 if (m_ui) m_ui->ProcessEvent(reinterpret_cast<const SDL::Event&>(ev));
135 if (m_scene) m_scene->DispatchInput(reinterpret_cast<const SDL::Event&>(ev));
136
137 if (ev.type == SDL::EVENT_KEY_DOWN) {
138 switch (ev.key.key) {
139 case SDL::KEYCODE_F1:
140 m_debugMode = !m_debugMode;
141 break;
142 case SDL::KEYCODE_ESCAPE:
143 m_paused = !m_paused;
144 m_ui->SetVisible(m_pausePanel, m_paused);
145 break;
146 case SDL::KEYCODE_RETURN:
147 case SDL::KEYCODE_KP_ENTER:
148 m_showMessage ? _AdvanceMessage() : _CheckFacingInteraction();
149 break;
150 case SDL::KEYCODE_KP_PLUS:
151 case SDL::KEYCODE_O:
152 m_dispTileSize = std::min(128, m_dispTileSize + 4);
153 m_camera.SetDispTileSize(m_dispTileSize); break;
154 case SDL::KEYCODE_KP_MINUS:
155 case SDL::KEYCODE_P:
156 m_dispTileSize = std::max(16, m_dispTileSize - 4);
157 m_camera.SetDispTileSize(m_dispTileSize); break;
158 default: break;
159 }
160 }
161 return true;
162 }

384
385 void _RenderGame(SDL::RendererRef r, SDL::FRect rect) {
386 // Clip all rendering to the canvas rect
387 r.SetViewport(FRectToRect(rect));
388
389 m_camera.SetViewport(0.f, 0.f, rect.w, rect.h);
390 _SyncSceneCamera(rect.w, rect.h); // ensure correct viewport for RenderScene
391
392 // 1. Background tile
393 _DrawBgTile(r);
394
395 // 2. Background layers
396 int bgEnd = m_map.numLayers - m_map.frontLayers;
397 _DrawTileLayers(r, 0, bgEnd);
398
399 // 3. Scene entities (Y-sorted via zOrder set by sync script)
400 SDL::ECS::RenderScene(m_gameWorld, r);
401
402 // 4. Foreground layers (draw over entities)
403 _DrawTileLayers(r, bgEnd, m_map.numLayers);
404
405 // 5. Per-entity health bars
406 _DrawEntityHUD(r);
407
408 // 6. Debug overlay
409 if (m_debugMode) _DrawDebug(r);
410
411 r.SetViewport(std::nullopt); // restore full viewport for UI
412 }

414 // ── Tile rendering ────────────────────────────────────────────────────────
415
416 void _DrawBgTile(SDL::RendererRef r) {
417 if (!m_tilesetTex || m_map.backgroundTile == 0) {
418 r.SetDrawColor({20,22,30,255});
419 r.RenderFillRect(SDL::FRect{0.f, 0.f, m_camera.viewW, m_camera.viewH});
420 return;
421 }
422 int id = m_map.backgroundTile;
423 if (m_map.zeroIsNotTile && id > 0) --id;
424 SDL::FRect src = _SrcRect(id);
425 float ts = (float)m_dispTileSize;
426 int sx, sy, ex, ey;
427 m_camera.VisibleTileRange(m_map.width, m_map.height, sx, sy, ex, ey);
428 for (int ty = sy; ty < ey; ++ty)
429 for (int tx = sx; tx < ex; ++tx) {
430 SDL::FRect dst = {m_camera.ScreenX((float)tx),
431 m_camera.ScreenY((float)ty), ts, ts};
432 r.RenderTexture(m_tilesetTex, src, dst);
433 }
434 }
435
436 void _DrawTileLayers(SDL::RendererRef r, int from, int to) {
437 if (!m_tilesetTex) return;
438 float ts = (float)m_dispTileSize;
439 int sx, sy, ex, ey;
440 m_camera.VisibleTileRange(m_map.width, m_map.height, sx, sy, ex, ey);
441 for (int layer = from; layer < to; ++layer) {
442 for (int ty = sy; ty < ey; ++ty) {
443 for (int tx = sx; tx < ex; ++tx) {
444 int id = m_map.GetTile(layer, tx, ty);
445 if (id == 0) continue;
446 int adj = (m_map.zeroIsNotTile && id > 0) ? id - 1 : id;
447 SDL::FRect src = _SrcRect(adj);
448 SDL::FRect dst = {m_camera.ScreenX((float)tx),
449 m_camera.ScreenY((float)ty), ts, ts};
450 r.RenderTexture(m_tilesetTex, src, dst);
451 }
452 }
453 }
454 }

456 SDL::FRect _SrcRect(int id) const noexcept {
457 float ts = (float)m_tilesetTileSize;
458 int col = id % m_tilesetCols;
459 int row = id / m_tilesetCols;
460 return {col * ts, row * ts, ts, ts};
461 }

463 // ── Per-entity health bars ────────────────────────────────────────────────
464 void _DrawEntityHUD(SDL::RendererRef r) {
465 m_gameWorld.Each<Transform, HealthComp, EntityTag>(
466 [&](SDL::ECS::EntityId eid, Transform& t, HealthComp& h, EntityTag&) {
467 if (!h.enabled || h.maxHp <= 0.f) return;

469 if (!vis->value) return;
470 float barW = (float)m_dispTileSize * 0.8f;
471 float bx = m_camera.ScreenX(t.x) - barW * 0.5f;
472 float by = m_camera.ScreenY(t.y) - (float)m_dispTileSize * 0.55f;
473 float pct = SDL::Clamp(h.hp / h.maxHp, 0.f, 1.f);
474 SDL::Color fg = pct > 0.5f ? SDL::Color{60,200,80,220}
475 : pct > 0.25f ? SDL::Color{220,180,40,220}
476 : SDL::Color{220,50,50,220};
477 SDL::FRect bg_r = {bx, by, barW, 5.f};
478 SDL::FRect fg_r = {bx, by, barW * pct, 5.f};
479 r.SetDrawColor({40,10,10,180}); r.RenderFillRect(bg_r);
480 r.SetDrawColor(fg); r.RenderFillRect(fg_r);

558 void _SpawnEntities() {
559 // Remove all non-player entities (scene + game components)
560 std::vector<SDL::ECS::EntityId> toRemove;
561 m_gameWorld.Each<EntityTag>([&](SDL::ECS::EntityId eid, EntityTag& tag) {
562 if (tag.type != EntityType::Player) toRemove.push_back(eid);
563 });
564 for (auto e : toRemove) m_scene->DestroyNode(e);
565 m_playerEntity = SDL::ECS::NullEntity;
566
567 EntityBuilder builder(m_gameWorld, *m_scene, m_ctx->renderer,
568 *m_ctx->pool, m_infoEntities,
569 m_assetsBase, &m_dispTileSize);
570
571 for (auto& sp : m_map.spawns) {
572 auto eid = builder.BuildFromSpawn(sp, m_entitiesGroup);
573 if (eid == SDL::ECS::NullEntity) continue;
574 if (sp.typeName == "PLAYER") {
575 m_playerEntity = eid;
576 m_camera.worldX = sp.x;
577 m_camera.worldY = sp.y;
578 m_scene->Emit("PLAYER", "spawned");
579 }
580 }
581 LOG_GAME(core::LogLevel::Info)
582 << std::format("Spawned {} entities", m_map.spawns.size());
583
584 // ── Apply pending save data (overrides spawn positions) ───────────────
585 if (m_hasPendingSave && m_playerEntity != SDL::ECS::NullEntity) {
586 const auto& sd = m_pendingSave;
587 if (auto* t = m_gameWorld.Get<Transform>(m_playerEntity)) {
588 t->x = sd.playerX; t->y = sd.playerY;
589 }
590 if (auto* d = m_gameWorld.Get<DirectionComp>(m_playerEntity))
591 d->dir = sd.playerDir;
592 if (auto* h = m_gameWorld.Get<HealthComp>(m_playerEntity)) {
593 h->hp = sd.hp; h->maxHp = sd.maxHp;
594 }
595 if (auto* p = m_gameWorld.Get<PlayerTag>(m_playerEntity))
596 p->polypoints = sd.polypoints;
597 m_camera.worldX = sd.playerX;
598 m_camera.worldY = sd.playerY;
599 m_hasPendingSave = false;
600 LOG_GAME(core::LogLevel::Success) << "Save data applied";
601 }

605 // Teleportation
606 // ═════════════════════════════════════════════════════════════════════════
607
608 void _CheckTeleport() {
609 if (m_playerEntity == SDL::ECS::NullEntity) return;
610 auto* res = m_gameWorld.Get<CollisionResult>(m_playerEntity);
611 if (!res || res->facing != core::CollisionId::Transition) return;
612
613 auto* tr = m_gameWorld.Get<Transform>(m_playerEntity);
614 auto* dir = m_gameWorld.Get<DirectionComp>(m_playerEntity);
615 if (!tr || !dir) return;
616
617 float fx = tr->x, fy = tr->y;
618 switch (dir->dir) {
619 case Direction::North: fy -= 1.f; break;
620 case Direction::South: fy += 1.f; break;
621 case Direction::West: fx -= 1.f; break;
622 case Direction::East: fx += 1.f; break;
623 }
624 auto key = std::format("{}_{}", (int)std::round(fx), (int)std::round(fy));
625 auto it = m_map.teleports.find(key);
626 if (it == m_map.teleports.end()) return;
627
628 const auto& tele = it->second;
629 if (tele.changeMap) {
630 LoadMap(tele.mapName);
631 }
632 // Reposition player (may be a new entity after LoadMap)
633 if (m_playerEntity != SDL::ECS::NullEntity)
634 if (auto* t2 = m_gameWorld.Get<Transform>(m_playerEntity)) {
635 t2->x = tele.x; t2->y = tele.y;
636 }
637 m_scene->Emit("WORLD", "teleport");
638 }

644 void _CheckFacingInteraction() {
645 if (m_playerEntity == SDL::ECS::NullEntity) return;
646 auto* tr = m_gameWorld.Get<Transform>(m_playerEntity);
647 auto* dir = m_gameWorld.Get<DirectionComp>(m_playerEntity);
648 auto* res = m_gameWorld.Get<CollisionResult>(m_playerEntity);
649 if (!tr || !dir || !res) return;
650 if (res->facing != core::CollisionId::Message &&
651 res->facing != core::CollisionId::Access) return;
652
653 float fx = tr->x, fy = tr->y;
654 switch (dir->dir) {
655 case Direction::North: fy -= 1.f; break;
656 case Direction::South: fy += 1.f; break;
657 case Direction::West: fx -= 1.f; break;
658 case Direction::East: fx += 1.f; break;
659 }
660 auto key = std::format("{}_{}", (int)std::round(fx), (int)std::round(fy));
661 auto it = m_map.messages.find(key);
662 if (it == m_map.messages.end()) return;
663
664 m_msgLines = it->second;
665 m_msgIdx = 0;
666 _ShowMsg(m_msgLines.empty() ? "" : m_msgLines[0]);
667 }
