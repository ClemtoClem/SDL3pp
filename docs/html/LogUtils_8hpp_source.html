137inline std::string GetTimestamp() {
138 using namespace std::chrono;
139 auto now = system_clock::now();
140 auto tt = system_clock::to_time_t(now);
141 auto ms = duration_cast<milliseconds>(now.time_since_epoch()) % 1000;
142
143 std::tm tmBuf{};
144#if defined(_WIN32)
145 localtime_s(&tmBuf, &tt);
146#else
147 localtime_r(&tt, &tmBuf);
148#endif
149 std::ostringstream ss;
150 ss << std::put_time(&tmBuf, "%H:%M:%S")
151 << '.' << std::setfill('0') << std::setw(3) << ms.count();
152 return ss.str();
153}

159class LogRingBuffer {
160public:
161 explicit LogRingBuffer(size_t capacity = LOG_RING_CAPACITY)
162 : m_capacity(capacity) {}
163
164 void Push(const LogEntry& e) {
165 std::lock_guard lk(m_mx);
166 if (m_buf.size() >= m_capacity) m_buf.pop_front();
167 m_buf.push_back(e);
168 }
169
171 [[nodiscard]] std::vector<LogEntry> GetRecent(size_t n = 16) const {
172 std::lock_guard lk(m_mx);
173 size_t count = std::min(n, m_buf.size());
174 std::vector<LogEntry> out;
175 out.reserve(count);
176 auto it = m_buf.end() - (std::ptrdiff_t)count;
177 out.assign(it, m_buf.end());
178 return out;
179 }
180
181 void Clear() { std::lock_guard lk(m_mx); m_buf.clear(); }
182 [[nodiscard]] size_t Size() const { std::lock_guard lk(m_mx); return m_buf.size(); }
183
184private:
185 mutable std::mutex m_mx;
186 std::deque<LogEntry> m_buf;
187 size_t m_capacity;
188};
