105inline std::vector<int> ParseCSVRow(const std::string& line) {
106 std::vector<int> row;
107 std::istringstream ss(line);
108 std::string token;
109 while (std::getline(ss, token, ',')) {
110 if (token.empty()) continue;
111 try { row.push_back((int)std::stoul(token)); }
112 catch (...) { row.push_back(0); }
113 }
114 return row;
115}

129inline MapData LoadMap(const std::string& basePath, const std::string& mapName) {
130 MapData map;
131 map.name = mapName;
132
133 const std::string confPath = basePath + "/" + mapName + ".script";
134 const std::string mapPath = basePath + "/" + mapName + ".map";
135 const std::string cmapPath = basePath + "/" + mapName + ".cmap";
136
137 // ── 1. Parse .script ──────────────────────────────────────────────────────
138 ScriptSectionPtr script;
139 try {
140 script = ParseConfFile(confPath);
141 } catch (const std::exception& e) {
142 LOG_RESOURCE(core::LogLevel::Error) << "LoadMap script: " << e.what();
143 return map;
144 }
145
146 map.width = script->count("width") ? (*script)["width"].AsInt(1) : 1;
147 map.height = script->count("height") ? (*script)["height"].AsInt(1) : 1;
148 map.numLayers = script->count("tracings") ? (*script)["tracings"].AsInt(1): 1;
149 map.frontLayers = script->count("nbTracingsInFrontOf")
150 ? (*script)["nbTracingsInFrontOf"].AsInt(0) : 0;
151 map.backgroundTile = script->count("backgroundTile")
152 ? (*script)["backgroundTile"].AsInt(0) : 0;
153 map.zeroIsNotTile = script->count("zeroIsNotTile")
154 ? (*script)["zeroIsNotTile"].AsBool(false) : false;
155 map.musicKey = script->count("musicName")
156 ? (*script)["musicName"].AsString("") : "";
157
158 int collisionOffset = script->count("tileCollisionIdOffset")
159 ? (*script)["tileCollisionIdOffset"].AsInt(0) : 0;
160
161 // ── Entities ──────────────────────────────────────────────────────────────
162 if (script->count("configEntities")) {
163 auto entSec = (*script)["configEntities"].AsSection();
164 if (entSec) {
165 for (auto& [k, v] : *entSec) {
166 auto sec = v.AsSection();
167 if (!sec) continue;
168 EntitySpawnDef def;
169 // Accept both "type" and "name" fields; normalise to UPPERCASE
170 std::string rawType = v.Get("type").AsString("");
171 if (rawType.empty()) rawType = v.Get("name").AsString("UNKNOWN");
172 def.typeName = ToUpperCase(rawType);
173 def.x = v.Get("x").AsFloat(0.f);
174 def.y = v.Get("y").AsFloat(0.f);
175 def.dir = ParseDirection(v.Get("direction").AsString("SOUTH"));
176 map.spawns.push_back(std::move(def));
177 }
178 }
179 }
180
181 // ── Messages ─────────────────────────────────────────────────────────────
182 if (script->count("messages")) {
183 auto msgSec = (*script)["messages"].AsSection();
184 if (msgSec) {
185 for (auto& [pos, lines] : *msgSec) {
186 auto lineSec = lines.AsSection();
187 if (!lineSec) continue;
188 std::vector<std::string> msgs;
189 for (int i = 1; ; ++i) {
190 auto it = lineSec->find('%' + std::to_string(i));
191 if (it == lineSec->end()) break;
192 msgs.push_back(it->second.AsString(""));
193 }
194 map.messages[pos] = std::move(msgs);
195 }
196 }
197 }
198
199 // ── Transitions — accepts both "transitions" and "teleportations" keys ────
200 {
201 const char* teleKeys[] = { "transitions", "teleportations" };
202 for (const char* key : teleKeys) {
203 if (!script->count(key)) continue;
204 auto teleSec = (*script)[key].AsSection();
205 if (!teleSec) continue;
206 for (auto& [pos, v] : *teleSec) {
207 TeleportDef def;
208 def.changeMap = v.Get("changeMap").AsBool(false);
209 // Accept both "name" and "type" as the destination map identifier
210 def.mapName = v.Get("name").AsString("");
211 if (def.mapName.empty()) def.mapName = v.Get("type").AsString("");
212 def.x = v.Get("x").AsFloat(0.f);
213 def.y = v.Get("y").AsFloat(0.f);
214 map.teleports[pos] = std::move(def);
215 }
216 break; // use the first key found
217 }
218 }
219
220 // ── 2. Parse .map ────────────────────────────────────────────────────────
221 {
222 std::ifstream f(mapPath);
223 if (!f.is_open()) {
224 LOG_RESOURCE(core::LogLevel::Error) << "Cannot open " << mapPath;
225 return map;
226 }
227
228 map.tiles.resize((size_t)map.numLayers,
229 std::vector<int>((size_t)(map.width * map.height), 0));
230
231 int currentLayer = 0;
232 int currentRow = 0;
233 std::string line;
234 while (std::getline(f, line)) {
235 // Windows line endings
236 if (!line.empty() && line.back() == '\r') line.pop_back();
237
238 if (line.empty()) {
239 // Layer separator
240 currentLayer++;
241 currentRow = 0;
242 continue;
243 }
244
245 if (currentLayer >= map.numLayers || currentRow >= map.height) continue;
246
247 auto row = detail::ParseCSVRow(line);
248 for (int x = 0; x < map.width && x < (int)row.size(); ++x) {
249 map.tiles[(size_t)currentLayer]
250 [(size_t)(currentRow * map.width + x)] = row[(size_t)x];
251 }
252 currentRow++;
253 }
254 }
255
256 // ── 3. Parse .cmap ────────────────────────────────────────────────────────
257 {
258 std::ifstream f(cmapPath);
259 if (!f.is_open()) {
260 LOG_RESOURCE(core::LogLevel::Error) << "Cannot open " << cmapPath;
261 // return partial map (no collision)
262 } else {
263 map.collision.resize((size_t)(map.width * map.height), 0);
264
265 int row = 0;
266 std::string line;
267 while (std::getline(f, line) && row < map.height) {
268 if (!line.empty() && line.back() == '\r') line.pop_back();
269 if (line.empty()) continue;
270
271 auto vals = detail::ParseCSVRow(line);
272 for (int x = 0; x < map.width && x < (int)vals.size(); ++x) {
273 int v = (int)vals[(size_t)x];
274 int normalized = (v > 0 && v >= collisionOffset) ? v - collisionOffset : 0;
275 map.collision[(size_t)(row * map.width + x)] =
276 (int)std::clamp(normalized, 0, 15);
277 }
278 row++;
279 }
280 }
281 }
282
283 LOG_RESOURCE(core::LogLevel::Success)
284 << "Map '" << mapName << "' loaded: "
285 << map.width << 'x' << map.height
286 << " | " << map.numLayers << " layers"
287 << " | " << map.spawns.size() << " entities";
288
289 return map;
290}
