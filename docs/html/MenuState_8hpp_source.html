133 void _DrawBackground(SDL::RendererRef r, SDL::FRect rect, SDL::Color acc) {
134 // Gradient background
135 for (int i = 0; i < (int)rect.h; ++i) {
136 float t = (float)i / rect.h;
137 r.SetDrawColorFloat(SDL::FColor{t*0.3f, t*0.3f, t*0.6f, 1.f});
138 r.RenderFillRect(SDL::FRect{rect.x, rect.y + rect.h * t, rect.w, rect.h / 12.f});
139 }
140
141 // Shimmer band
142 float lx = rect.x + rect.w * (0.5f + 0.5f * SDL::Sin(m_time * 0.4f));
143 r.SetDrawColor({acc.r, acc.g, acc.b, 70});
144 r.RenderFillRect(SDL::FRect{lx - 80.f, rect.y, 160.f, rect.h});
145
146 // Bottom border
147 r.SetDrawColor({acc.r, acc.g, acc.b, 220});
148 r.RenderFillRect(SDL::FRect{rect.x, rect.y + rect.h - 2.f, rect.w, 2.f});
149
150 // Title (4×)
151 float pct = 0.5f + 0.5f * SDL::Sin(m_time * 0.5f);
152 r.SetDrawColor({(uint8_t)(180 + 75 * pct),
153 (uint8_t)(180 + 75 * SDL::Sin(m_time*0.5f+2.09f)),
154 255, 255});
155 const char* title = "POLYADVENTURE";
156 float tw = SDL::Strlen(title) * 8.f * 4.f;
157 r.SetScale({4.f, 4.f});
158 r.RenderDebugText({(rect.x + rect.w * 0.5f - tw * 0.5f) / 4.f,
159 (rect.y + rect.h * 0.38f - 16.f) / 4.f}, title);
160 r.SetScale({1.f, 1.f});
161
162 // Subtitle (1.5×)
163 r.SetDrawColor({160, 165, 200, 200});
164 const char* sub = "An SDL3pp C++20 RPG";
165 float sw = SDL::Strlen(sub) * 8.f * 1.5f;
