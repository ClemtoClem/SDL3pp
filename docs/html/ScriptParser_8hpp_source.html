119struct ScriptSection {
120 using Pair = std::pair<std::string, ScriptValue>;
121 using Entries = std::vector<Pair>;
122 using Index = std::map<std::string, std::size_t>;
123
124 Entries entries;
125 Index index;
126
127 void Set(std::string key, ScriptValue val) {
128 auto it = index.find(key);
129 if (it != index.end()) {
130 entries[it->second].second = std::move(val);
131 } else {
132 index.emplace(key, entries.size());
133 entries.emplace_back(std::move(key), std::move(val));
134 }
135 }
136
137 [[nodiscard]] std::size_t count(const std::string& key) const noexcept {
138 return index.count(key);
139 }
140 ScriptValue& operator[](const std::string& key) {
141 auto it = index.find(key);
142 if (it == index.end()) {
143 index.emplace(key, entries.size());
144 entries.emplace_back(key, ScriptValue{});
145 return entries.back().second;
146 }
147 return entries[it->second].second;
148 }
149 [[nodiscard]] const ScriptValue& operator[](const std::string& key) const noexcept {
150 auto it = index.find(key);
151 if (it == index.end()) { static const ScriptValue kNull; return kNull; }
152 return entries[it->second].second;
153 }
154
155 [[nodiscard]] Entries::iterator begin() noexcept { return entries.begin(); }
156 [[nodiscard]] Entries::iterator end() noexcept { return entries.end(); }
157 [[nodiscard]] Entries::const_iterator begin() const noexcept { return entries.begin(); }
158 [[nodiscard]] Entries::const_iterator end() const noexcept { return entries.end(); }
159 [[nodiscard]] Entries::const_iterator cbegin() const noexcept { return entries.cbegin();}
160 [[nodiscard]] Entries::const_iterator cend() const noexcept { return entries.cend(); }
161
162 [[nodiscard]] Entries::iterator find(const std::string& key) noexcept {
163 auto it = index.find(key);
164 return (it == index.end()) ? entries.end()
165 : entries.begin() + static_cast<std::ptrdiff_t>(it->second);
166 }

193// ── Low-level char-stream parser that writes into ObjectDataNode ──────────────

195class GameScriptParser {
196public:
197 explicit GameScriptParser(std::string src) noexcept : m_src(std::move(src)) {}
198
199 void ParseInto(SDL::ObjectDataNode& root) { _ParseBody(root, '\0'); }
200
201private:
202 std::string m_src;
203 std::size_t m_pos = 0;
204 int m_line = 1;
205
206 [[nodiscard]] char _Peek() const noexcept {
207 return m_pos < m_src.size() ? m_src[m_pos] : '\0';
208 }
209 char _Get() noexcept {
210 char c = _Peek();
211 if (c == '\n') ++m_line;
212 if (c) ++m_pos;
213 return c;
214 }
215 bool _At(char c) const noexcept { return _Peek() == c; }
216
217 void _SkipWS() noexcept {
218 while (m_pos < m_src.size()) {
219 char c = _Peek();
220 if (c == ' ' || c == '\t' || c == '\r') { ++m_pos; }
221 else if (c == '\n') { ++m_line; ++m_pos; }
222 else if (c == '#') { while (m_pos < m_src.size() && _Peek() != '\n') ++m_pos; }
223 else break;
224 }
225 }
226 bool _Expect(char c) noexcept { _SkipWS(); if (_At(c)) { _Get(); return true; } return false; }
227
228 // ── Key: identifier | %N | (x,y)→"x_y" ──────────────────────────────────
229 std::string _ParseKey() {
230 _SkipWS();
231 if (m_pos >= m_src.size()) return "";
232 char c = _Peek();
233
234 if (c == '%') {
235 std::string k; k += _Get();
236 while (m_pos < m_src.size() && std::isdigit(_Peek())) k += _Get();
237 return k;
238 }
239 if (c == '(') {
240 _Get();
241 std::string xs, ys;
242 while (m_pos < m_src.size() && _Peek() != ',' && _Peek() != ')') xs += _Get();
243 if (_At(',')) _Get();
244 while (m_pos < m_src.size() && _Peek() != ')') ys += _Get();
245 if (_At(')')) _Get();
246 auto trim = [](const std::string& s) {
247 auto a = s.find_first_not_of(" \t"), b = s.find_last_not_of(" \t");
248 return (a == std::string::npos) ? std::string{} : s.substr(a, b - a + 1);
249 };
250 return trim(xs) + "_" + trim(ys);
251 }
252 if (std::isalpha(c) || c == '_') {
253 std::string k;
254 while (m_pos < m_src.size() && (std::isalnum(_Peek()) || _Peek() == '_')) k += _Get();
255 return k;
256 }
257 return "";
258 }

260 // ── Rect args helper (handles 'f' suffix per value) ───────────────────────
261 static std::shared_ptr<SDL::ObjectDataNode> _BuildRect(const std::string& args) {
262 auto obj = SDL::ObjectDataNode::Make();
263 obj->set(kRectTag, SDL::BoolDataNode::Make(true));
264 float vals[4] = {};
265 int idx = 0;
266 std::istringstream ss(args);
267 std::string tok;
268 while (idx < 4 && std::getline(ss, tok, ',')) {
269 auto a = tok.find_first_not_of(" \t"), b = tok.find_last_not_of(" \t");
270 if (a != std::string::npos) tok = tok.substr(a, b - a + 1);
271 if (!tok.empty() && (tok.back() == 'f' || tok.back() == 'F')) tok.pop_back();
272 try { vals[idx] = std::stof(tok); } catch (...) {}
273 ++idx;
274 }
275 static const char* keys[4] = { kRectX, kRectY, kRectW, kRectH };
276 for (int i = 0; i < 4; ++i) obj->set(keys[i], SDL::F32DataNode::Make(vals[i]));
277 return obj;
278 }

280 // ── Value ─────────────────────────────────────────────────────────────────
281 std::shared_ptr<SDL::DataNode> _ParseValue() {
282 _SkipWS();
283 char c = _Peek();
284
285 if (c == '{') {
286 _Get();
287 auto obj = SDL::ObjectDataNode::Make();
288 _ParseBody(*obj, '}');
289 return obj;
290 }
291 if (c == '[') {
292 _Get();
293 return _ParseArray();
294 }
295 if (c == '"') return SDL::StringDataNode::Make(_ParseString());
296
297 return _ParseSimple();
298 }
299
300 std::string _ParseString() {
301 _Get(); // '"'
302 std::string s;
303 while (m_pos < m_src.size()) {
304 char c = _Get();
305 if (c == '"') break;
306 if (c == '\\' && m_pos < m_src.size()) {
307 char esc = _Get();
308 switch (esc) {
309 case 'n': s += '\n'; break; case 't': s += '\t'; break;
310 case '"': s += '"'; break; case '\\':s += '\\'; break;
311 default: s += '\\'; s += esc; break;
312 }
313 continue;
314 }
315 s += c;
316 }
317 return s;
318 }

320 std::shared_ptr<SDL::DataNode> _ParseSimple() {
321 _SkipWS();
322
323 // rect / frect — peek for identifier then '('
324 {
325 std::size_t savedPos = m_pos;
326 int savedLine = m_line;
327 std::string ident;
328 while (m_pos < m_src.size() && std::isalpha(_Peek())) ident += _Get();
329 if (!ident.empty() && _At('(')
330 && (ident == "rect" || ident == "frect")) {
331 _Get(); // '('
332 std::string args;
333 int depth = 1;
334 while (m_pos < m_src.size() && depth > 0) {
335 char ch = _Get();
336 if (ch == '(') { ++depth; args += ch; }
337 else if (ch == ')') { --depth; if (depth > 0) args += ch; }
338 else args += ch;
339 }
340 return _BuildRect(args);
341 }
342 m_pos = savedPos; m_line = savedLine;
343 }
344
345 // Plain token
346 std::string tok;
347 while (m_pos < m_src.size()) {
348 char ch = _Peek();
349 if (ch == ' ' || ch == '\t' || ch == '\n' || ch == '\r'
350 || ch == '=' || ch == '{' || ch == '}' || ch == ']'
351 || ch == ';' || ch == '#')
352 break;
353 tok += _Get();
354 }
355 if (tok.empty()) return SDL::NoneDataNode::Make();
356
357 if (tok == "true" || tok == "yes") return SDL::BoolDataNode::Make(true);
358 if (tok == "false" || tok == "no") return SDL::BoolDataNode::Make(false);
359
360 // Float with 'f' suffix
361 if (tok.back() == 'f' || tok.back() == 'F') {
362 std::string num(tok.begin(), tok.end() - 1);
363 try { return SDL::F32DataNode::Make(std::stof(num)); } catch (...) {}
364 }
365 // Float with '.'
366 if (tok.find('.') != std::string::npos) {
367 try { return SDL::F32DataNode::Make(std::stof(tok)); } catch (...) {}
368 }
369 // Integer
370 try { return SDL::S32DataNode::Make(std::stoi(tok)); } catch (...) {}
371
372 return SDL::StringDataNode::Make(tok);
373 }
374
375 // ── Body (key=value pairs) ────────────────────────────────────────────────
376 void _ParseBody(SDL::ObjectDataNode& node, char terminator) {
377 while (true) {
378 _SkipWS();
379 if (m_pos >= m_src.size()) break;
380 if (terminator != '\0' && _At(terminator)) { _Get(); break; }
381
382 std::string key = _ParseKey();
383 if (key.empty()) { while (m_pos < m_src.size() && _Peek() != '\n') _Get(); continue; }
384
385 _SkipWS();
386 if (!_Expect('=')) { while (m_pos < m_src.size() && _Peek() != '\n') _Get(); continue; }
387
388 node.set(key, _ParseValue());
389 }
390 }
391
392 // ── Array (anonymous items) ───────────────────────────────────────────────
393 std::shared_ptr<SDL::ArrayDataNode> _ParseArray() {
394 auto arr = SDL::ArrayDataNode::Make();
395 int idx = 0;
396 while (true) {
397 _SkipWS();
398 if (m_pos >= m_src.size() || _At(']')) { if (_At(']')) _Get(); break; }
399 if (_At('{')) {
400 _Get();
401 auto sub = SDL::ObjectDataNode::Make();
402 _ParseBody(*sub, '}');
403 arr->add(sub);
404 } else if (_At('"')) {
405 arr->add(SDL::StringDataNode::Make(_ParseString()));
406 } else {
407 arr->add(_ParseSimple());
408 }
409 (void)idx++;
410 }
411 return arr;
412 }

456inline ScriptValue BridgeValue(const std::shared_ptr<SDL::DataNode>& n) {
457 if (!n) return ScriptValue{};
458 using DT = SDL::DataNodeType;
459 switch (n->getType()) {
460 case DT::NONE: return ScriptValue{};
461 case DT::BOOLEAN: return ScriptValue(std::dynamic_pointer_cast<SDL::BoolDataNode>(n)->getValue());
462 case DT::S8: return ScriptValue((int)std::dynamic_pointer_cast<SDL::S8DataNode>(n)->getValue());
463 case DT::U8: return ScriptValue((int)std::dynamic_pointer_cast<SDL::U8DataNode>(n)->getValue());
464 case DT::S16: return ScriptValue((int)std::dynamic_pointer_cast<SDL::S16DataNode>(n)->getValue());
465 case DT::U16: return ScriptValue((int)std::dynamic_pointer_cast<SDL::U16DataNode>(n)->getValue());
466 case DT::S32: return ScriptValue((int)std::dynamic_pointer_cast<SDL::S32DataNode>(n)->getValue());
467 case DT::U32: return ScriptValue((int)std::dynamic_pointer_cast<SDL::U32DataNode>(n)->getValue());
468 case DT::S64: return ScriptValue((int)std::dynamic_pointer_cast<SDL::S64DataNode>(n)->getValue());
469 case DT::U64: return ScriptValue((int)std::dynamic_pointer_cast<SDL::U64DataNode>(n)->getValue());
470 case DT::F32: return ScriptValue(std::dynamic_pointer_cast<SDL::F32DataNode>(n)->getValue());
471 case DT::F64: return ScriptValue((float)std::dynamic_pointer_cast<SDL::F64DataNode>(n)->getValue());
472 case DT::STRING: return ScriptValue(std::dynamic_pointer_cast<SDL::StringDataNode>(n)->getValue());
473 case DT::ARRAY: {
474 auto arr = std::dynamic_pointer_cast<SDL::ArrayDataNode>(n);
475 auto sec = std::make_shared<ScriptSection>();
476 for (std::size_t i = 0; i < arr->getSize(); ++i)
477 sec->Set(std::to_string(i), BridgeValue(arr->get(i)));
478 return ScriptValue(sec);
479 }
480 case DT::OBJECT: {
481 auto obj = std::dynamic_pointer_cast<SDL::ObjectDataNode>(n);
482 // Rect objects carry a "__rect" sentinel bool
483 if (obj->has(kRectTag)) {
484 auto getF = [&](const char* k) -> float {
485 auto sub = obj->get(k);
486 if (!sub) return 0.f;
487 if (auto p = std::dynamic_pointer_cast<SDL::F32DataNode>(sub)) return p->getValue();
488 if (auto p = std::dynamic_pointer_cast<SDL::S32DataNode>(sub)) return (float)p->getValue();
489 return 0.f;
490 };
491 return ScriptValue(ScriptRect{getF(kRectX), getF(kRectY), getF(kRectW), getF(kRectH)});
492 }
493 return ScriptValue(BridgeSection(obj));
494 }
495 default: return ScriptValue{};
496 }
497}

499inline ScriptSectionPtr BridgeSection(const std::shared_ptr<SDL::ObjectDataNode>& n) {
500 auto sec = std::make_shared<ScriptSection>();
501 if (!n) return sec;
502 for (const auto& [k, v] : n->getValues())
503 sec->Set(k, BridgeValue(v));
504 return sec;
505}
