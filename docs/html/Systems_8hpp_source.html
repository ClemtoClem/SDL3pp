26// Convert a box in source-frame pixels to ecs_context-unit bounds centred on (cx, cy).
27// boxTileSize = source frame size (e.g. 64 px per tile in the sprite sheet)
28// ecs_contextScale = 1 ecs_context unit = ecs_contextScale source pixels
29inline SDL::FBox BoxToWorld(const SDL::FRect& b, float cx, float cy, int boxTileSize) noexcept {
30 float half = boxTileSize * 0.5f;
31 float left = cx + (b.x - half) / boxTileSize;
32 float top = cy + (b.y - half) / boxTileSize;
33 float right = left + b.w / boxTileSize;
34 float bot = top + b.h / boxTileSize;
35 return { left, top, right, bot };
36}

44 [&](SDL::ECS::EntityId, PlayerTag&, Transform&,
45 Velocity& vel, DirectionComp& dir, SpeedComp& spd, SpriteAnim& anim) {
46
47 float speed = spd.running ? spd.runSpeed : spd.walkSpeed;
48 vel.vx = 0.f; vel.vy = 0.f;
49
50 if (keys[SDL::SCANCODE_LEFT] || keys[SDL::SCANCODE_A]) { vel.vx = -speed; dir.dir = Direction::West; }
51 if (keys[SDL::SCANCODE_RIGHT] || keys[SDL::SCANCODE_D]) { vel.vx = speed; dir.dir = Direction::East; }
52 if (keys[SDL::SCANCODE_UP] || keys[SDL::SCANCODE_W]) { vel.vy = -speed; dir.dir = Direction::North; }
53 if (keys[SDL::SCANCODE_DOWN] || keys[SDL::SCANCODE_S]) { vel.vy = speed; dir.dir = Direction::South; }

55 anim.moving = (vel.vx != 0.f || vel.vy != 0.f);
56
57 // Diagonal normalization
58 if (vel.vx != 0.f && vel.vy != 0.f) {
59 float inv = speed / SDL::Sqrt(2.f);
60 vel.vx = (vel.vx > 0) ? inv : -inv;
61 vel.vy = (vel.vy > 0) ? inv : -inv;
62 }
63 });

70inline void Collision(Context& ecs_context, const core::MapData& map) {
71 ecs_context.Each<Transform, Velocity, DirectionComp, CollisionBoxes>(
72 [&](SDL::ECS::EntityId, Transform& t, Velocity& vel,
73 DirectionComp& dir, CollisionBoxes& boxes) {
74
75 auto& step = boxes.step[(int)dir.dir];
76 int ts = boxes.tileSize;
77
78 auto isTileBlocking = [&](float cx, float cy) {
79 auto box = BoxToWorld(step, cx, cy, ts);
80 for (int ty = (int)SDL::Floor(box.top); ty <= (int)SDL::Floor(box.bottom); ++ty) {
81 for (int tx = (int)SDL::Floor(box.left); tx <= (int)SDL::Floor(box.right); ++tx) {
82 if (core::IsBlocking(map.GetCollision(tx, ty))) return true;
83 }
84 }
85 return false;
86 };
87
88 // Try X movement
89 if (vel.vx != 0.f && !isTileBlocking(t.x + vel.vx, t.y))
90 t.x += vel.vx;
91
92 // Try Y movement
93 if (vel.vy != 0.f && !isTileBlocking(t.x, t.y + vel.vy))
94 t.y += vel.vy;
95
96 // Clamp to map bounds
97 t.x = std::clamp(t.x, 0.5f, (float)map.width - 0.5f);
98 t.y = std::clamp(t.y, 0.5f, (float)map.height - 0.5f);
99 });
100}

147inline void MobAISystem(Context& ecs_context, float dt) {
148 // Find player position
149 float playerX = -1.f, playerY = -1.f;
150 SDL::ECS::EntityId playerId = SDL::ECS::NullEntity;
151 ecs_context.Each<PlayerTag, Transform>([&](SDL::ECS::EntityId eid, PlayerTag&, Transform& pt) {
152 playerX = pt.x; playerY = pt.y; playerId = eid;
153 });
154
155 ecs_context.Each<MobAI, Transform, Velocity, DirectionComp, SpriteAnim>(
156 [&](SDL::ECS::EntityId, MobAI& ai, Transform& t,
157 Velocity& vel, DirectionComp& dir, SpriteAnim& anim) {
158
159 ai.stateTimer = std::max(0.f, ai.stateTimer - dt);
160 ai.attackCooldown = std::max(0.f, ai.attackCooldown - dt);
161 vel.vx = vel.vy = 0.f;
162 anim.moving = false;
163
164 if (ai.state == AIState::Dead) return;
165
166 float d2 = (playerX >= 0) ? dist2(t.x, t.y, playerX, playerY) : 1e9f;
167
168 switch (ai.state) {
169 case AIState::Idle: {
170 if (ai.stateTimer <= 0.f) {
171 ai.state = AIState::Patrol;
172 ai.stateTimer = 2.f + (float)(rand() % 300) / 100.f;
173 float angle = (float)(rand() % 628) / 100.f;
174 ai.patrolTargX = ai.spawnX + std::cos(angle) * ai.patrolRadius;
175 ai.patrolTargY = ai.spawnY + std::sin(angle) * ai.patrolRadius;
176 }
177 // Fall through to chase check
178 if (playerX >= 0 && d2 < ai.chaseRadius * ai.chaseRadius)
179 ai.state = AIState::Chase;
180 break;
181 }
182 case AIState::Patrol: {
183 if (playerX >= 0 && d2 < ai.chaseRadius * ai.chaseRadius) {
184 ai.state = AIState::Chase; break;
185 }
186 float dx = ai.patrolTargX - t.x, dy = ai.patrolTargY - t.y;
187 float d = std::sqrt(dx*dx + dy*dy);
188 if (d < 0.2f || ai.stateTimer <= 0.f) {
189 ai.state = AIState::Idle;
190 ai.stateTimer = 1.f;
191 } else {
192 float inv = 0.025f / d;
193 vel.vx = dx * inv; vel.vy = dy * inv;
194 anim.moving = true;
195 if (std::abs(dx) > std::abs(dy))
196 dir.dir = (dx > 0) ? Direction::East : Direction::West;
197 else
198 dir.dir = (dy > 0) ? Direction::South : Direction::North;
199 }
200 break;
201 }
202 case AIState::Chase: {
203 if (playerX < 0 || d2 > (ai.chaseRadius * 1.5f) * (ai.chaseRadius * 1.5f)) {
204 ai.state = AIState::Idle; ai.stateTimer = 1.f; break;
205 }
206 if (d2 < ai.attackRange * ai.attackRange) {
207 ai.state = AIState::Attack; break;
208 }
209 float dx = playerX - t.x, dy = playerY - t.y;
210 float d = std::sqrt(dx*dx + dy*dy);
211 float inv = 0.025f / d;
212 vel.vx = dx * inv; vel.vy = dy * inv;
213 anim.moving = true;
214 if (std::abs(dx) > std::abs(dy))
215 dir.dir = (dx > 0) ? Direction::East : Direction::West;
216 else
217 dir.dir = (dy > 0) ? Direction::South : Direction::North;
218 ai.targetEntity = playerId;
219 break;
220 }
221 case AIState::Attack: {
222 if (playerX < 0 || d2 > ai.attackRange * ai.attackRange * 4.f) {
223 ai.state = AIState::Chase; break;
224 }
225 if (ai.attackCooldown <= 0.f) {
226 // Deal damage to player
227 if (ecs_context.IsAlive(playerId)) {
228 if (auto* ph = ecs_context.Get<HealthComp>(playerId)) {
229 if (!ph->sick && ph->invincibleTimer <= 0.f) {
230 ph->hp = SDL::Max(0.f, ph->hp - 5.f);
231 ph->invincibleTimer = 0.5f;
232 }
233 }
234 }
235 ai.attackCooldown = 1.5f;
236 }
237 break;
238 }
239 default: break;
240 }
241 });
242}

247
248inline void Health(Context& ecs_context, float dt) {
249 ecs_context.Each<HealthComp>([&](SDL::ECS::EntityId, HealthComp& h) {
250 if (!h.enabled) return;
251 h.invincibleTimer = std::max(0.f, h.invincibleTimer - dt);
252 if (h.sick)
253 h.hp = std::max(0.f, h.hp - h.damagePerSec * dt);
254 else if (h.regenPerSec > 0.f)
255 h.hp = std::min(h.maxHp, h.hp + h.regenPerSec * dt);

265 [&](SDL::ECS::EntityId, AutoWalk& aw, Transform& t,
266 Velocity& vel, DirectionComp& dir, SpriteAnim& anim) {
267 if (!aw.active) { vel.vx = vel.vy = 0.f; return; }
268 float dx = aw.targetX - t.x, dy = aw.targetY - t.y;
269 float d = std::sqrt(dx*dx + dy*dy);
270 if (d < 0.05f) { aw.active = false; vel.vx = vel.vy = 0.f; return; }
271 float spd = 0.025f;

273 vel.vy = dy / d * spd;
274 anim.moving = true;

276 dir.dir = (dx > 0) ? Direction::East : Direction::West;
277 else
278 dir.dir = (dy > 0) ? Direction::South : Direction::North;
